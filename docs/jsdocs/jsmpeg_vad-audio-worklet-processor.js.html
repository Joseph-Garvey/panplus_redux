        this.var.lastBin = 80;//MUST be 80. Otherwise, LINE_COEFFS must be updated
        this.var.fft1BinCount = 512;
        this.var.fft1 = new VADAudioWorkletProcessor.FFTJS(this.var.fft1BinCount);
        this.var.fft1Out = this.var.fft1.createComplexArray();
        this.var.fft2BinCount = 128;
        this.var.fft2 = new VADAudioWorkletProcessor.FFTJS(this.var.fft2BinCount);
        this.var.fft2In = new Float32Array(this.var.fft2BinCount);
        this.var.fft2Out = this.var.fft2.createComplexArray();
        this.var.buffer = [new Float32Array(this.var.fft1BinCount), new Float32Array(this.var.fft1BinCount)];
        this.var.bufferIndex = 0;
        this.var.continueBuffering = true;
        this.var.speakingHistory = 0;
        this.var.silenceTimeThreshold = 1;
        this.var.logGraphData = {};
        this.var.sampleRate = 48000;
        this.var.startProcessingFrom = 0;
        this.var.lagTime = this.calculateLagTime();
        this.var.timePerInterval = 128 / this.var.sampleRate;
        this.var.currentTSTimeElapsed = 0;
        this.var.currentTSIndex = 0;
        this.var.processedTimes = 0;
        //Other variables but initialized below
        //this.var.metadataArray = [];

        //this.referenceLineFeatures = null;

        this.port.onmessage = (function(event) {
            switch (event.data.msgEnum) {

    /**
     * Inputs of length 128, buffer size of 512
     * https://en.wikipedia.org/wiki/Voice_frequency "In telephony, the usable voice frequency band ranges from approximately 300 Hz to 3400 Hz."
     * https://en.wikipedia.org/wiki/Sampling_(signal_processing)#Speech_sampling
     * Refer to https://webaudio.github.io/web-audio-api/#dom-audioworkletprocessor-process for the actual API
     * @param {Float32Array} inputs inputs[n][m] is a Float32Array of audio samples for the mth channel of the nth input.
     * @param {Float32Array} outputs The output audio buffer that is to be consumed by the user agent
     * @param {Object} parameters A map of string keys and associated Float32Arrays. parameters["name"] corresponds to the automation values of the AudioParam named "name".
     * @returns {undefined}
     */
    process(inputs, outputs, parameters) {
        this.var.processedTimes++;
        if (this.errorOccurred()) return true;
        this.var.currentTSTimeElapsed += this.var.timePerInterval;
        const metadata = this.var.metadataArray[this.var.currentTSIndex];
        if (this.var.currentTSTimeElapsed > metadata.duration) {
            //Eating into the other TS file
            //Reset and move on
            this.var.currentTSTimeElapsed = 0;
            //Add new results array for next TS file
            this.var.results.push({data: [], metadata: metadata});
            if (++this.var.currentTSIndex >= this.var.metadataArray.length) {
                //Finished last TS file, terminate
                return false;
            }
            //Clear buffer
            this.var.bufferIndex = 0;
        } else if (this.var.currentTSTimeElapsed >= metadata.startProcessingFrom) {
            if (this.var.msgEnum == this.var.MessageEnums.NOISE_RESULTS) { 
                this.bufferThenProcess(inputs, outputs, parameters, this.processReference.bind(this));
            } 
            else { 
                this.bufferThenProcess(inputs, outputs, parameters, this.processData.bind(this)); 
            }
        }
        return true;
    }

    /**
     * Buffer samples for rectangle windowing
     * Note: buffering means the last few pieces of info in a TS file may be dropped if it doesn't mod 512 
     * @param {Float32Array} inputs The input audio buffer from the incoming connections provided by the user agent.
     * @param {Float32Array} outputs The output audio buffer that is to be consumed by the user agent
     * @param {Object} parameters A map of string keys and associated Float32Arrays. parameters["name"] corresponds to the automation values of the AudioParam named "name".
     * @param {Function} calledFunction function that is called with the buffer as an argument
     * @returns {undefined}
     */
    bufferThenProcess(inputs, outputs, parameters, calledFunction) {
        if (this.var.continueBuffering) {
            if (this.var.bufferIndex >= this.var.fft1BinCount) {
                calledFunction(parameters);
                //"clear" for reuse buffer
                this.var.bufferIndex = 0;
            }

            //concat to buffer, use set because this is Float32Array
            for (var i = 0; i < inputs[0].length; i++) {
                this.var.buffer[i].set(inputs[0][i], this.var.bufferIndex);
            }
            this.var.bufferIndex += inputs[0][0].length;    
        }
    }

    /**
     * Process & average some samples of noise (assume that the start of the webcast is noise) as reference.
     * @param {Object} parameters A map of string keys and associated Float32Arrays. parameters["name"] corresponds to the automation values of the AudioParam named "name".
     * @returns {undefined}
     */
    processReference(parameters) {
        //this.log("refResults", this.results);
        if (this.var.referenceSamples == null) {
            this.var.referenceSamples = 0;
            this.var.referenceData = [];
            this.var.results = [];
            for (let i = 0; i < 2; i++) {
                this.var.results.push([]);
                this.var.referenceData.push([]);
                for (let j = 0; j < 5; j++) {
                    this.var.referenceData[i].push([]);
                    this.var.results[i].push({mean: 0, stddev: 0});
                }
            }
        }
        if (this.var.referenceSamples < this.var.noiseSampleCount) {
            let newData = this.processData(parameters, false);
            this.var.referenceSamples++;
            for (let i = 0; i < this.var.referenceData.length; i++) {
                for (let j = 0; j < this.var.referenceData[i].length; j++) {
                    ////Push to array of data
                    this.var.referenceData[i][j].push(newData[i][j]);
                    //Sum for mean later
                    this.var.results[i][j].mean += newData[i][j];
                }
            }
            //Gathered enough samples
            if (this.var.referenceSamples >= this.var.noiseSampleCount) {
                //Calculate mean and std dev of each variable
                for (let i = 0; i < this.var.referenceData.length; i++) {
                    for (let j = 0; j < this.var.referenceData[i].length; j++) {
                        //Calculate mean
                        this.var.results[i][j].mean /= this.var.referenceData[i][j].length;
                        //Calculate variance & subsequently std dev
                        for (let k = 0; k < this.var.referenceData[i][j].length; k++) {
                            this.var.results[i][j].stddev += Math.pow(this.var.referenceData[i][j][k] - this.var.results[i][j].mean, 2);
                        }
                        this.var.results[i][j].stddev /= this.var.referenceData[i][j].length - 1;
                        this.var.results[i][j].stddev = Math.sqrt(this.var.results[i][j].stddev);
                        //Calculate the final bounds for each variable
                        this.var.results[i][j] = {
                            lowerBound: this.var.results[i][j].mean - this.var.results[i][j].stddev * this.var.stddevDistance[j], 
                            upperBound: this.var.results[i][j].mean + this.var.results[i][j].stddev * this.var.stddevDistance[j]
                        };
                    }
                }
                this.var.continueBuffering = false;
            }
        }
    }

    /**
     * VAD section
     * https://www.ncbi.nlm.nih.gov/pmc/articles/PMC4142156/
     * @param {Boolean} checkForSpeech Check if speaking against reference.
     * @returns {Array.<Array.<Number>>} array of line features, usually 2 line features of 5 variables
     */
    processData(parameters, checkForSpeech = true) {
        //process for both channels
        let lineFeatures = [];
        let isSpeaking = false;
        for (let i = 0; i < this.var.buffer.length; i++) {
            //DCFT, but with 48k sample instead of downsampled 8k.
            //Don't want to downsample to reduce noise created from downsampling & reduce processing.
            //Instead, first FFT will FFT the entire 48k sample, then 2nd FFT will use the range from 0 - 12k Hz.
            //First FFT
            //this.log("buffer", this.buffer[i]);
            this.var.fft1.realTransform(this.var.fft1Out, this.var.buffer[i]);
            //this.var.log("fft1Out", this.var.fft1Out);
            //Prepare magnitude of first 128 bins (0 - 12k Hz)
            this.binsToMagnitudeArray(this.var.fft1Out, this.var.fft2BinCount, this.var.fft2In);
            //this.var.log("fft2In1", [this.var.fft2Out, this.var.fft2In]);
            //Second FFT to better distinguish the harmonic signals from 0 Hz - 12k Hz
            this.var.fft2.realTransform(this.var.fft2Out, this.var.fft2In);
            //this.var.log("fft2Out", [this.var.fft2Out, this.var.fft2In]);
            //Reuse the fft2In buffer to store our magnitudes
            this.binsToMagnitudeArray(this.var.fft2Out, this.var.lastBin, this.var.fft2In);
            //this.var.log("fft2In2", this.var.fft2In);
            lineFeatures.push(this.calculateLineFeatures(this.var.fft2In, this.var.lastBin));
            //Check against reference noise if is speaking
            /*//Implementation 1: static threshold
            if (checkForSpeech && this.var.calculateDistanceMeasurement(lineFeatures[i], this.var.referenceLineFeatures[i]) > this.var.distanceThreshold) {
                isSpeaking = true;
                break;
            }*/
            /*//Implementation 2: debugging
            if (checkForSpeech) {
                this.var.recordDistance(this.var.calculateDistanceMeasurement(lineFeatures[i], this.var.referenceLineFeatures[i]));
            }*/
            if (checkForSpeech && this.distanceOutsideBounds(lineFeatures[i], this.var.referenceLineFeatures[i])) {
                isSpeaking = true;
                break;
            }
        }

        //Implementation that involves checking for speech in the worklet
        if (isSpeaking) {
            if (this.var.speakingHistory === 0) {
                this.changedSpeech(true);
            }
            this.var.speakingHistory = this.var.silenceTimeThreshold;
        } else if (this.var.speakingHistory > 0) {
            this.var.speakingHistory--;
            if (this.var.speakingHistory === 0) {
                this.changedSpeech(false);
            }
        }
        return lineFeatures;
    }

    /**
     * Use to convert a complex array to an array of magnitudes
     * @param {Array} complexArray Refer to FFT.js documentation
     * @param {Number} binCount number of bins
     * @param {Array} magArr Array of magnitudes
     * @returns {undefined} Passed by reference
     */
    binsToMagnitudeArray(complexArray, binCount, magArr) {
        //this.log("binsToMagnitudeArray", [complexArray, binCount, magArr]);
        for (let i = 0; i < binCount; i++) {
            magArr[i] = this.calculateFftBinMagnitude(complexArray, i);
            //this.log("midBins", [magArr, i, magArr[i]]);
        }
        //this.log("postBinsToMagnitudeArray", [complexArray, binCount, magArr]);
    }

    /**
     * Use to calculate line features (see https://www.ncbi.nlm.nih.gov/pmc/articles/PMC4142156/)
     * @param {Array} magArr Array of magnitudes
     * @param {Number} magArrLength 
     * @returns {Array.<Number>} line feature, contains 5 variables/coefficients 
     */
    calculateLineFeatures(magArr, magArrLength) {
        let lineFeatures = [];
        //Get Fmean and L
        let sumMagDivj = 0;
        let sumMagLog2JDivj = 0;
        let sumMag = 0;
        for (let j = 1; j <= magArrLength; j++) {
            //ERROR: CALCULATION ERROR HERE, LINE_COEFFS are SUM
            sumMagDivj += magArr[j] * this.var.LINE_COEFFS.jReciprocal[j];
            sumMagLog2JDivj += magArr[j] * this.var.LINE_COEFFS.log2jOverj[j];
            sumMag += magArr[j];
        }
        //lineFeatures.fMean
        if (sumMag != 0 || sumMagDivj != 0)
            lineFeatures.push(sumMag / sumMagDivj);
        else {//This is not entirely correct though
            //But for our purposes I'll assume 0 / 0 = 0 when magArr is completely 0 
            lineFeatures.push(0);
        }
        let l = Math.floor(lineFeatures[0]);
        //Calculate low index line fitting coeffs
        //Matrix values
        let a = this.var.SUMMED_COEFFS.jReciprocal[l];
        let bOrC = this.var.SUMMED_COEFFS.log2jOverj[l];
        let d = this.var.SUMMED_COEFFS.log2jSqOverj[l];
        let determinant = 1 / (a * d - bOrC * bOrC);
        if (determinant == Infinity) {
            //This is also not entirely correct either
            //but for our purposes I'll assume Infinity / 0 = 0
            determinant = 0;
        }
        let x = 0;
        let y = 0;
        for (let j = 1; j <= l; j++) {
            x += magArr[j] * this.var.LINE_COEFFS.jReciprocal[j];
            y += magArr[j] * this.var.LINE_COEFFS.log2jOverj[j];
        }
        //lineFeatures.al0 
        lineFeatures.push(determinant * (a * x + bOrC * y));
        //lineFeatures.al1 
        lineFeatures.push(determinant * (bOrC * x + d * y));
        for (let i = 0; i < lineFeatures.length; i++) {
            if (!isFinite(lineFeatures[i])) {
                this.log("NaN LF Before", {magArr: magArr, lineFeatures: lineFeatures, sumMag: sumMag, sumMagDivj: sumMagDivj, sumMagLog2JDivj: sumMagLog2JDivj, determinant: determinant, a: a, bOrC: bOrC, d: d, x: x, y: y,l: l, magArrLength: magArrLength});
                this.logSend(this.var.MessageEnums.ERROR);
            }
        }
        //Calculate high index line fitting coeffs
        a = this.var.SUMMED_COEFFS.jReciprocal[magArrLength] - a;
        bOrC = this.var.SUMMED_COEFFS.log2jOverj[magArrLength] - bOrC;
        d = this.var.SUMMED_COEFFS.log2jSqOverj[magArrLength] - d;
        determinant = 1 / (a * d - bOrC * bOrC);
        x = sumMagDivj - x;
        y = sumMagLog2JDivj - y;
        //lineFeatures.ah0 
        lineFeatures.push(determinant * (a * x + bOrC * y));
        //lineFeatures.ah1 
        lineFeatures.push(determinant * (bOrC * x + d * y));

        for (let i = 0; i < lineFeatures.length; i++) {
            if (!isFinite(lineFeatures[i])) {
                this.log("NaN LF After", {magArr: magArr, lineFeatures: lineFeatures, sumMag: sumMag, sumMagDivj: sumMagDivj, sumMagLog2JDivj: sumMagLog2JDivj, determinant: determinant, a: a, bOrC: bOrC, d: d, x: x, y: y,l: l, magArrLength: magArrLength});
                this.logSend(this.var.MessageEnums.ERROR);
            }
        }

        return lineFeatures;
    }

    /**
     * Calculate the distance between sets of two line features. Idea is that the line features of a noise signal will be similar to that of the reference signal.
     * @deprecated
     * @param {Array.<Number>} lineFeatures1 Line features 1 of length 5
     * @param {Array.<Number>} lineFeatures2 Line features 2 of length 5
     * @returns {Number} distance
     */
    calculateDistanceMeasurement(lineFeatures1, lineFeatures2) {
        let sum = 0;
        for (let i = 0; i < lineFeatures1.length; i++) {
            sum += Math.pow(lineFeatures1[i] - lineFeatures2[i], 2);
        }
        sum = +Math.sqrt(sum).toFixed(1);
        //this.addToGraphData(sum);
        //this.log("dist", sum);
        return sum;
    }

    /**
     * Check if the line features are within the "boundaries" of the noise reference signal
     * @param {Array.<Number>} lineFeatures Line features of length 5 (variables that define the signal)
     * @param {Array.<{Number, Number}>} boundsArr Array containing the lower and upper bounds of each line feature variable
     * @returns {Boolean} exceeded bounds
     */
    distanceOutsideBounds(lineFeatures, boundsArr) {
        let varExceeded = 0;
        for (let i = 0; i < lineFeatures.length; i++) {
            if (lineFeatures[i] < boundsArr[i].lowerBound || lineFeatures[i] > boundsArr[i].upperBound) {
                varExceeded++;
                if (varExceeded >= this.var.variablesFailBeforeSignificant) break;
            }
        }
        return varExceeded >= this.var.variablesFailBeforeSignificant;
    }

    /**
     * Calculate the magnitude for a specific FFT bin
     * @param {Array} complexArray Refer to FFT.js documentation
     * @param {Number} binIndex bin index in complex array
     * @returns {Number}
     */
    calculateFftBinMagnitude(complexArray, binIndex) {
        ////this.log("calculateFftBinMagnitude", [complexArray, binIndex, binIndex*2, complexArray[binIndex]*complexArray[binIndex] + complexArray[binIndex+1]*complexArray[binIndex+1], Math.sqrt(complexArray[binIndex]*complexArray[binIndex] + complexArray[binIndex+1]*complexArray[binIndex+1])]);
        binIndex *= 2;
        return Math.sqrt(complexArray[binIndex]*complexArray[binIndex] + complexArray[binIndex+1]*complexArray[binIndex+1]);
    }

    /**
     * Calculate the lag time in ms (basically the time spent buffering data to process it)
     * @returns {Number}
     */
    calculateLagTime() {
        return this.var.fft1BinCount / this.var.sampleRate;
    }

    /**
     * The threshold is now variant. Using recordDistance instead
     * Store into results upon changing speech
     * @param {Boolean} startedSpeech isSpeaking
     * @returns {undefined}
     */
    changedSpeech(startedSpeech) {
        this.var.results[this.var.currentTSIndex].data.push({
            isSpeaking: startedSpeech, 
            time: this.var.metadataArray[this.var.currentTSIndex].startTime + this.var.currentTSTimeElapsed - this.var.lagTime
        });
    }

    /**
     * Used for logging purposes to communicate debug data from processor
     * @param {*} key 
     * @param {*} value 
     * @returns {undefined}
     */
    log(key, value) {
        //Define self so to avoid collision with "this\.log"
        let self = this;
        if (self.var.logData == null) {
            self.var.logData = [];
        }
        self.var.logData.push({t: currentTime, k: key, v: value});
    }

    /**
     * Used to send debug logs (e.g. in the event of a crash).
     * @param {MessageEnums} definedEnum
     * @returns {undefined}
     */
    logSend(definedEnum = this.MessageEnums.DEBUG) {
        //Define self so to avoid collision with "this\.log"
        let self = this;
        if (self.var.logData == null) {
            self.var.logData = [];
        }
        this.var.sentLog = true;
        this.port.postMessage({data: self.var.logData, msgEnum: definedEnum});
    }

    /**
     * @returns {Boolean} if logSend was called
     */
    logBeenSent() {
        return this.var.sentLog === true;
    }

    /**
     * In the event that logSend was called with MesssageEnums.ERROR, the worklet will cease processing the TS.
     * @returns {Boolean}
     */
    errorOccurred() {
        return this.var.sentError;
    }
}
